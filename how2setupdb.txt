Prefix table:
CREATE TABLE public.prefixes
(
    prefix text COLLATE pg_catalog."default" NOT NULL,
    owner_id bigint NOT NULL,
    CONSTRAINT prefixes_pkey PRIMARY KEY (owner_id)
